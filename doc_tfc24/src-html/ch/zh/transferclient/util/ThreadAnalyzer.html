001package ch.zh.transferclient.util;
002
003/**
004 * The class is used for thread analyzing purposes.
005 * 
006 * @author  Daniel Bierer (Statistisches Amt des Kantons ZÃ¼rich)
007 * @version 2.4
008 *
009 */
010public class ThreadAnalyzer
011    
012    {
013    
014    /**
015     * Constructs a ThreadAnalyzer object.
016     */
017    private ThreadAnalyzer()
018        {
019        //see also https://stackoverflow.com/questions/31409982/java-best-practice-class-with-only-static-methods
020        }
021    
022    /**
023     * Flag which indicates whether the thread analyzer is active or not.
024     */
025    private static final boolean ACTIVE = false;
026    
027    /**
028     * Prints out information about a thread and the corresponding task on which the thread is working.
029     * 
030     * @param thread        Thread which is under observation.
031     * @param thread_type   Type of the thread.
032     * @param task_type     Type of the task.
033     * @param details       Additional information.
034     */
035    public static void println(Thread thread, String thread_type, String task_type, String details)
036        {
037        
038        if (ACTIVE)
039            {
040            System.out.printf("%-45s %-35s %-25s %s \n", thread, thread_type, task_type, details);
041            }
042        }
043        
044    /**
045     *  Prints out the header of the protocol.
046     */
047    public static void println_head()
048        {
049        if (ACTIVE)
050            {
051            System.out.printf("-----------------------------------------------------------------------------------------------------------------------------------------\n");
052            System.out.printf("%-45s %-35s %-25s %s \n", "Thread", "Thread Type", "Task Type", "Details");
053            System.out.printf("-----------------------------------------------------------------------------------------------------------------------------------------\n");
054            }
055        }
056        
057    }
