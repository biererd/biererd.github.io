001package ch.zh.transferclient.util;
002
003public class ThreadAnalyzer
004    
005    {
006    
007    public static final boolean ACTIVE = false;
008    
009    public static void println(Thread thread, String thread_type, String task_type, String details)
010        {
011        
012        if (ACTIVE)
013            {
014            System.out.printf("%-45s %-35s %-25s %s \n", thread, thread_type, task_type, details);
015            }
016        }
017        
018    public static void println_head()
019        {
020        if (ACTIVE)
021            {
022            System.out.printf("-----------------------------------------------------------------------------------------------------------------------------------------\n");
023            System.out.printf("%-45s %-35s %-25s %s \n", "Thread", "Thread Type", "Task Type", "Details");
024            System.out.printf("-----------------------------------------------------------------------------------------------------------------------------------------\n");
025            }
026        }
027        
028    }
