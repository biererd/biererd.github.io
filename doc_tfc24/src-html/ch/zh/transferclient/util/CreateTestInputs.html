001package ch.zh.transferclient.util;
002
003import java.io.BufferedWriter;
004import java.io.FileOutputStream;
005import java.io.OutputStreamWriter;
006
007import ch.zh.transferclient.main.Logger;
008
009/**
010 * 
011 * The class is used for simulating input files.
012 * 
013 * @author  Daniel Bierer (Statistisches Amt des Kantons Zürich)
014 * @version 2.4
015 *
016 */
017public class CreateTestInputs
018    
019    {
020    
021    /**
022     * Constructs a CreateTestInputs object.
023     */
024    private CreateTestInputs()
025        {
026        //see also https://stackoverflow.com/questions/31409982/java-best-practice-class-with-only-static-methods
027        }
028    
029    /**
030     * Writes 1000 input files to the input folder.
031     * 
032     * @param args Command-line arguments.
033     */
034    public static void main(String[] args)
035        
036        {
037        
038        for (int i = 0; i < 1000; i++)
039            {
040            String id;
041            if (i < 10)
042                {
043                id = "000" + i;
044                }
045            else if (i < 100)
046                {
047                id = "00" + i;
048                }
049            else if (i < 1000)
050                {
051                id = "0" + i;
052                }
053            else
054                {
055                id = "" + i;
056                }
057                
058            write_file("TEST_TRANSFER_CLIENT_VERSION_2_4_FILE_" + id + ".txt");
059            }
060            
061        }
062    
063    
064    /**
065     * Writes a test input file.
066     * 
067     * @param filename The file name to be used.
068     */
069    private static void write_file(String filename)
070        {
071        
072        String dir = "c://0_tf_reliability//files_fuer_stresstest_5//";
073        
074        try
075            {
076            
077            BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(dir + filename)));
078            
079            bw.write("---------------------------------------------------------");
080            bw.newLine();
081            bw.write(" BEIM VORLIEGENDEN FILE HANDELT ES SICH UM EIN TESTFILE, ");
082            bw.newLine();
083            bw.write(" DAS VERSENDET WORDEN IST, UM DIE NEUE VERSION 2.4       ");
084            bw.newLine();
085            bw.write(" DES TRANSFER-CLIENTS ZU TESTEN.                         ");
086            bw.newLine();
087            bw.write("---------------------------------------------------------");
088            bw.newLine();
089            bw.write(" Absender:                                               ");
090            bw.newLine();
091            bw.write(" Statistisches Amt des Kantons Zürich                    ");
092            bw.newLine();
093            bw.write(" Supportdienst Transfer-Client                           ");
094            bw.newLine();
095            bw.write(" Tel.: 043 259 75 23                                     ");
096            bw.newLine();
097            bw.write("---------------------------------------------------------");
098            bw.newLine();
099            
100            bw.flush();
101            bw.close();
102            
103            }
104        catch (Exception e)
105            {
106            Logger.error(e);
107            }
108            
109        }
110        
111    }
