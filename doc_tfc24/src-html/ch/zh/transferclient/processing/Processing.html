017package ch.zh.transferclient.processing;
018
019import java.io.*;
020
021import java.nio.file.Files;
022import java.nio.file.Path;
023import java.nio.file.Paths;
024
025import javax.swing.SwingUtilities;
026
027import ch.zh.transferclient.gui.*;
028import ch.zh.transferclient.main.Logger;
029import ch.zh.transferclient.properties.Properties;
030import ch.zh.transferclient.util.TimeStamp;
031
032/**
033 * This class is used to process the input and receipt files.
034 *
035 * @author  Daniel Bierer (Statistical Office of the Canton of Zurich)
036 * @version 2.4
037 */
038public class Processing
039    
040    {
041    
042    /**
043     * Processes the input and receipt files.
044     * 
045     * @param properties The properties to be used.
046     * @param gui        The graphical user interface to be used.
047     */
048    public synchronized static void process(final Properties properties, final Gui gui)
049        
050        {
051        
052        final String sedex_sender_id = properties.get_sedex_sender_id();
053        final String dir_results     = properties.get_folder_results();
054        
055        // -------------------------------------------------//
056        // Schlaufe ueber alle Files des Resultate-Ordners //
057        // -------------------------------------------------//
058        final File   dir             = new File(dir_results);
059        
060        //Identifizierung der Files
061        //There is no guarantee that the name strings in the resulting array will appear in any specific order; 
062        //they are not, in particular, guaranteed to appear in alphabetical order.
063        //https://docs.oracle.com/javase/7/docs/api/java/io/File.html#listFiles()
064        final File[] files           = dir.listFiles();
065        
066        //Damit der Executor-Thread bei der Deaktivierung
067        //auch bei vielen gleichzeitigen Versandanfragen 
068        //schnell unterbrochen wird, wird neu nur noch 
069        //das erste File der Liste verarbeitet.
070        //Die Verarbeitung der folgenden Files erfolgt
071        //dann im folgenden Taskdurchlauf.
072        int size = files.length;
073        if (size>0)
074            {
075            size=1;
076            }
077        
078        //for (int i = 0; i < files.length; i++)
079        for (int i = 0; i < size; i++)
080            {
081            if (files[i].isDirectory())
082                {
083                // Verzeichnisse werden nicht weggeschoben
084                }
085            else
086                {
087                final File   file             = files[i];
088                final String datafile_path    = file.getAbsolutePath();
089                final Path   path_source      = Paths.get(datafile_path);
090                final String sedex_message_id = sedex_sender_id + "-" + TimeStamp.getstamp_for_sedex_message_id();
091                
092                // Einbau eines Delays von 100 Millisekunden zur Gewährleistung,
093                // dass eine Message-ID nicht zweimal vergeben werden kann.
094                // https://docs.oracle.com/javase/tutorial/essential/concurrency/sleep.html
095                try
096                    {
097                    Thread.sleep(100);
098                    }
099                catch (Exception e)
100                    {
101                    Logger.error(e);
102                    SwingUtilities.invokeLater(new Runnable()
103                        {
104                        @Override
105                        public void run()
106                            {
107                            gui.get_dialog_fileprocessingerror().setvisible(e);
108                            }
109                        });
110                    }
111                    
112                try
113                    {
114                    
115                    // In der nachfolgenden Schlaufe wird so lange gewartet,
116                    // bis das File wegkopiert werden kann.
117                    // Exceptions koennen hier umso haeufiger auftreten,
118                    // je groesser die Datenfiles sind.
119                    // Wenn beispielsweise 100 MB grosse Files in den Resultateordner
120                    // geschrieben werden, kann es gut sein, dass diese noch
121                    // fuer eine laengere Zeit blockiert sind.
122                    // Diese Schlaufe wurde mit der Version 2.2 des
123                    // Transfer-Clients eingefuehrt.
124                    while (true)
125                        {
126                        try
127                            {
128                            Files.copy(file.toPath(), new File("archive/data/data_" + sedex_message_id + "_"
129                                    + file.getName()).toPath());
130                            break;
131                            }
132                        catch (Exception e)
133                            {
134                            }
135                        if (!files[i].exists())
136                            break;
137                        }
138                        
139                    // Falls die Datenfiles nicht archiviert werden sollen,
140                    // wird das File nachfolgend wieder geloescht.
141                    // Wieso wird das File nicht einfach nur dann wegkopiert,
142                    // wenn es archiviert werden soll?
143                    // Begruendung:
144                    // Durch das gewaehlte Vorgehen ist sicher gestellt,
145                    // dass bei der nachfolgenden Weiterverarbeitung das
146                    // File vom externen System fertig geschrieben ist
147                    // (man denke an groessere Files >100 MB).
148                    boolean archive_datafiles = properties.get_archive_datafiles();
149                    if (!archive_datafiles)
150                        {
151                        Files.deleteIfExists(new File("archive/data/data_" + sedex_message_id + "_"
152                                + file.getName()).toPath());
153                        }
154                        
155                    // ------------------------------//
156                    // Verarbeitung des Input-Files //
157                    // ------------------------------//
158                    Processing_SingleInput.process(properties, gui, file, sedex_message_id);
159                    
160                    // ---------------------//
161                    // Receipts-Processing //
162                    // ---------------------//
163                    Processing_Receipts.process_receipts(properties, gui);
164                    
165                    Logger.info("FILE PROCESSED: " + sedex_message_id + ": " + path_source.toString());
166                    
167                    }
168                catch (Exception e)
169                    {
170                    Logger.error(e);
171                    SwingUtilities.invokeLater(new Runnable()
172                        {
173                        @Override
174                        public void run()
175                            {
176                            gui.get_dialog_fileprocessingerror().setvisible(e);
177                            }
178                        });
179                    }
180                    
181                }
182                
183            }
184            
185        // @formatter:off                                                                    //
186        // ----------------------------------------------------------------------------------//
187        // Receipts-Processing                                                               //
188        // ----------------------------------------------------------------------------------//
189        // Wieso findet das Receipt-Processing hier nochmals statt, obwohl es in der         //
190        // obigen Schlaufe ueber die Inputfiles bereits aufgefuehrt ist?                     //
191        // Begruendung:                                                                      //
192        // Falls keine Input-Files mehr vorhanden sind, wuerde das Receipts-Processing       //
193        // nicht mehr durchgefuehrt werden.                                                  //
194        // ----------------------------------------------------------------------------------//
195        // Wieso findet das Receipt-Processing ueberhaupt auch noch in der Schleife          //
196        // statt?                                                                            //
197        // Begruendung:                                                                      //
198        // Angenommen, ein Benutzer legt 1000 Files auf einmal ins Verzeichnis, dann         //
199        // dauert es sehr lange, bis das Receipt-Processing einsetzt, naemlich erst nach     //
200        // nach dem Versenden aller 1000 Files.                                              //
201        // ----------------------------------------------------------------------------------//
202        // @formatter:on                                                                     //
203        try
204            {
205            Processing_Receipts.process_receipts(properties, gui);
206            }
207        catch (InterruptedException | java.lang.reflect.InvocationTargetException e)
208            {
209            Logger.error(e);
210            SwingUtilities.invokeLater(new Runnable()
211                {
212                @Override
213                public void run()
214                    {
215                    gui.get_dialog_fileprocessingerror().setvisible(e);
216                    }
217                });
218            }
219            
220        }
221        
222    }
