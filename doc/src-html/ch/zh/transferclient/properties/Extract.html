017package ch.zh.transferclient.properties;
018
019import java.io.BufferedInputStream;
020import java.io.BufferedReader;
021import java.io.FileInputStream;
022import java.io.InputStreamReader;
023import java.nio.charset.StandardCharsets;
024
025/**
026 * This class is used to extract the properties from the properties file.
027 * 
028 * @author  Daniel Bierer (Statistical Office of the Canton of Zurich)
029 * @version 2.4
030 */
031public class Extract
032    
033    {
034    
035    /** Indicator which indicates whether the files should be compressed or not. */
036    private final String zip_compression;
037    
038    /**
039     * Indicator which indicates whether the log should be overwritten for every new session or not.
040     */
041    private final String log_overwrite;
042    /**
043     * Indicator which indicates whether debug messages in the log should be suppressed or not.
044     */
045    private final String log_suppress_debug;
046    /**
047     * Indicator which indicates whether transmitted files should be archived or not.
048     */
049    private final String archive_datafiles;
050    
051    /** Language to be used in the graphical user interface. */
052    private final String language;
053    
054    /** Sedex ID of the sender. */
055    private final String sedex_sender_id;
056    /** Sedex IDs of the recipients. */
057    private final String sedex_recipient_id;
058    /** Sedex folder. */
059    private final String sedex_dir;
060    /** Sedex outbox folder. */
061    private final String sedex_dir_outbox;
062    /** Sedex receipts folder. */
063    private final String sedex_dir_receipts;
064    
065    /** The folder of the result files. */
066    private final String folder_results;
067    /** The time when the transfer-client should be automatically activated. */
068    private final String target_time;
069    
070    /** The number of table entries. */
071    private final String number_of_table_entries;
072        
073    /**
074     * @return Indicator which indicates whether the files should be compressed or not.
075     */
076    protected String get_zip_compression()
077        {
078        return this.zip_compression;
079        }
080        
081    /**
082     * @return Indicator which indicates whether the log should be overwritten for every new session or not.
083     */
084    protected String get_log_overwrite()
085        {
086        return this.log_overwrite;
087        }
088        
089    /**
090     * @return Indicator which indicates whether debug messages in the log should be suppressed or not.
091     */
092    protected String get_log_suppress_debug()
093        {
094        return this.log_suppress_debug;
095        }
096        
097    /**
098     * @return Indicator which indicates whether transmitted files should be archived or not.
099     */
100    protected String get_archive_datafiles()
101        {
102        return this.archive_datafiles;
103        }
104        
105    /** @return Language to be used in the graphical user interface. */
106    protected String get_language()
107        {
108        return this.language;
109        }
110        
111    /** @return Sedex ID of the sender. */
112    protected String get_sedex_sender_id()
113        {
114        return this.sedex_sender_id;
115        }
116        
117    /** @return Sedex IDs of the recipients. */
118    protected String get_sedex_recipient_id()
119        {
120        return this.sedex_recipient_id;
121        }
122        
123    /** @return Sedex folder. */
124    protected String get_sedex_dir()
125        {
126        return this.sedex_dir;
127        }
128        
129    /** @return Sedex outbox folder. */
130    protected String get_sedex_dir_outbox()
131        {
132        return this.sedex_dir_outbox;
133        }
134        
135    /** @return Sedex receipts folder. */
136    protected String get_sedex_dir_receipts()
137        {
138        return this.sedex_dir_receipts;
139        }
140        
141    /** @return The folder of the result files. */
142    protected String get_folder_results()
143        {
144        return this.folder_results;
145        }
146        
147    /**
148     * @return The time when the transfer-client should be automatically activated.
149     */
150    protected String get_target_time()
151        {
152        return this.target_time;
153        }
154        
155    /** @return The number of table entries. */
156    protected String get_number_of_table_entries()
157        {
158        return this.number_of_table_entries;
159        }
160        
161    /**
162     * Extracts the properties from the properties file.
163     * 
164     * @throws Exception Exception thrown if the properties cannot be extracted.
165     */
166    protected Extract() throws Exception
167        {
168        
169        String              zip_compression                 = "";
170        
171        String              log_overwrite                   = "";
172        String              log_suppress_debug              = "";
173        String              archive_datafiles               = "";
174        
175        String              language                        = "";
176        
177        String              sedex_sender_id                 = "";
178        String              sedex_recipient_id              = "";
179        String              sedex_dir                       = "";
180        String              sedex_dir_outbox                = "";
181        String              sedex_dir_receipts              = "";
182        
183        String              folder_results                  = "";
184        String              target_time                     = "";
185        
186        String              number_of_table_entries         = "";
187        
188        // Identifizierung des Ausführungsverzeichnisses
189        // Das Properties-File wir im gleichen Verzeichnis gesucht,
190        // wo sich das zur Ausführung gebrachte JAR-File befindet.
191        String              dir_ausfuehrung                 = System.getProperty("user.dir");
192        
193        BufferedInputStream stream                          = new BufferedInputStream(new FileInputStream(dir_ausfuehrung
194                + "/properties.txt"));
195        
196        BufferedReader      r                               = new BufferedReader(new InputStreamReader(stream, StandardCharsets.UTF_8));
197        
198        String              line;
199        while ((line = r.readLine()) != null)
200            {
201            
202            if (!line.trim().startsWith("#"))
203                {
204                if (line.contains("="))
205                    {
206                    String[] paare     = line.split("=");
207                    
208                    String   parameter = paare[0].toLowerCase().trim();
209                    
210                    if (parameter.equals("zip_compression"))
211                        {
212                        if (paare.length > 1)
213                            {
214                            zip_compression = paare[1].trim();
215                            }
216                        }
217                    else if (parameter.equals("log_overwrite"))
218                        {
219                        if (paare.length > 1)
220                            {
221                            log_overwrite = paare[1].trim();
222                            }
223                        }
224                    else if (parameter.equals("log_suppress_debug"))
225                        {
226                        if (paare.length > 1)
227                            {
228                            log_suppress_debug = paare[1].trim();
229                            }
230                        }
231                    else if (parameter.equals("archive_datafiles"))
232                        {
233                        if (paare.length > 1)
234                            {
235                            archive_datafiles = paare[1].trim();
236                            }
237                        }
238                    else if (parameter.equals("language"))
239                        {
240                        if (paare.length > 1)
241                            {
242                            language = paare[1].trim();
243                            }
244                        }
245                    else if (parameter.equals("sedex_sender_id"))
246                        {
247                        if (paare.length > 1)
248                            {
249                            sedex_sender_id = paare[1].trim();
250                            }
251                        }
252                    else if (parameter.equals("sedex_recipient_id"))
253                        {
254                        if (paare.length > 1)
255                            {
256                            sedex_recipient_id = paare[1].trim();
257                            }
258                        }
259                    else if (parameter.equals("sedex_dir"))
260                        {
261                        
262                        if (paare.length > 1)
263                            {
264                            
265                            // Warum nicht replaceAll()? in der folgenden Zeile?
266                            // Grund:
267                            // https://stackoverflow.com/questions/6805028/how-to-replace-backward-slash-to-forward-slash-using-java
268                            sedex_dir = paare[1].replace("\\", "/").trim();
269                            }
270                        }
271                        
272                    else if (parameter.equals("sedex_dir_outbox"))
273                        {
274                        
275                        if (paare.length > 1)
276                            {
277                            
278                            // Warum nicht replaceAll()? in der folgenden Zeile?
279                            // Grund:
280                            // https://stackoverflow.com/questions/6805028/how-to-replace-backward-slash-to-forward-slash-using-java
281                            sedex_dir_outbox = paare[1].replace("\\", "/").trim();
282                            }
283                        }
284                        
285                    else if (parameter.equals("sedex_dir_receipts"))
286                        {
287                        
288                        if (paare.length > 1)
289                            {
290                            
291                            // Warum nicht replaceAll()? in der folgenden Zeile?
292                            // Grund:
293                            // https://stackoverflow.com/questions/6805028/how-to-replace-backward-slash-to-forward-slash-using-java
294                            sedex_dir_receipts = paare[1].replace("\\", "/").trim();
295                            }
296                        }
297                        
298                    else if (parameter.equals("folder_results"))
299                        
300                        {
301                        if (paare.length > 1)
302                            {
303                            
304                            // Warum nicht replaceAll()? in der folgenden Zeile?
305                            // Grund:
306                            // https://stackoverflow.com/questions/6805028/how-to-replace-backward-slash-to-forward-slash-using-java
307                            folder_results = paare[1].replace("\\", "/").trim();
308                            }
309                        }
310                        
311                    else if (parameter.equals("target_time"))
312                        {
313                        if (paare.length > 1)
314                            {
315                            target_time = paare[1].trim();
316                            }
317                        }
318                        
319                    else if (parameter.equals("number_of_table_entries"))
320                        {
321                        if (paare.length > 1)
322                            {
323                            number_of_table_entries = paare[1].trim();
324                            }
325                        }
326                        
327                    }
328                    
329                }
330                
331            }
332            
333        r.close();
334        stream.close();
335        
336        this.zip_compression                 = zip_compression;
337        
338        this.log_overwrite                   = log_overwrite;
339        this.log_suppress_debug              = log_suppress_debug;
340        this.archive_datafiles               = archive_datafiles;
341        
342        this.language                        = language;
343        
344        this.sedex_sender_id                 = sedex_sender_id;
345        this.sedex_recipient_id              = sedex_recipient_id;
346        this.sedex_dir                       = sedex_dir;
347        this.sedex_dir_outbox                = sedex_dir_outbox;
348        this.sedex_dir_receipts              = sedex_dir_receipts;
349        
350        this.folder_results                  = folder_results;
351        this.target_time                     = target_time;
352        
353        this.number_of_table_entries         = number_of_table_entries;
354        
355        }
356        
357    }
